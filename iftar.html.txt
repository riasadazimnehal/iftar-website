<script>
let todayIftar;

function getLocation(){
    if(navigator.geolocation){
        navigator.geolocation.getCurrentPosition(fetchPrayerTime);
    } else {
        alert("Geolocation not supported.");
    }
}

function fetchPrayerTime(position){
    const lat = position.coords.latitude;
    const lon = position.coords.longitude;

    document.getElementById("location").innerHTML =
        "Lat: " + lat.toFixed(2) + " | Lon: " + lon.toFixed(2);

    fetch(`https://api.aladhan.com/v1/timings?latitude=${lat}&longitude=${lon}&method=1`)
    .then(response => response.json())
    .then(data => {
        todayIftar = data.data.timings.Maghrib;
        document.getElementById("iftarTime").innerHTML =
            "Today Iftar: " + todayIftar;
        startCountdown();
    });
}

function startCountdown(){
    setInterval(() => {
        const now = new Date();
        const [hour,minute] = todayIftar.split(":");

        const iftar = new Date();
        iftar.setHours(hour);
        iftar.setMinutes(minute);
        iftar.setSeconds(0);

        const diff = iftar - now;

        if(diff > 0){
            const h = Math.floor(diff / (1000*60*60));
            const m = Math.floor((diff%(1000*60*60))/(1000*60));
            const s = Math.floor((diff%(1000*60))/1000);

            document.getElementById("countdown").innerHTML =
                h+"h "+m+"m "+s+"s left";
        }else{
            document.getElementById("countdown").innerHTML =
                "Iftar Time! üåô";
        }
    },1000);
}

getLocation();
</script>
